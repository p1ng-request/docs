### Create barlinechart using Plotly


```python
def create_linechart(
    df_n,
    df_n_1,
    xaxis_title=None,
    yaxis_title=None,
):
    # Create figure
    fig = go.Figure()

    # Add traces
    fig.add_trace(
        go.Scatter(
            name="N",
            x=df_n["LABEL"],
            y=df_n["VALUE"],
            marker=dict(color="blue"),
        )
    )
    fig.add_trace(
        go.Scatter(
            name="N-1",
            x=df_n_1["LABEL"],
            y=df_n_1["VALUE"],
            marker=dict(color="orange"),
        )
    )
    # Calc var
    last_n = df_n.loc[df_n.index[-1], "VALUE"]
    last_n_1 = df_n_1.loc[df_n_1.index[-1], "VALUE"]
    var = last_n - last_n_1
    if var > 0:
        var = f"+{var}"

    # Add figure title
    fig.update_layout(
        title=f"Ventes : {last_n} € ({var} € vs N-1)",
        title_font=dict(family="Arial", size=18, color="black"),
        legend=None,
        plot_bgcolor="#ffffff",
        width=1200,
        height=800,
        paper_bgcolor="white",
        xaxis_title=xaxis_title,
        xaxis_title_font=dict(family="Arial", size=12, color="black"),
        xaxis={"type": "category"},
    )

    # Set y-axes titles
    fig.update_yaxes(
        title_text=yaxis_title,
        title_font=dict(family="Arial", size=12, color="black"),
    )
    fig.show()
    return fig


fig = create_linechart(
    df_n,
    df_n_1,
    xaxis_title="Mois",
    yaxis_title="Valeur en €",
)
```
