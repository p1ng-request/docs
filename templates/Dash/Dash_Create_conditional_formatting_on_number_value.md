### Create app layout


```python
app.title = "Naas"
app.layout = html.Div(
    [
        html.H3("Enter your number value"),
        html.P("If value = 0 then color = orange, if value > 0 then green, if value < 0 then red"),
        dcc.Input(id="input"),
        html.H4(id="value", children="Color")
    ],
    style={
        "padding": "100px"
    }
)

@app.callback(
    Output("value", "style"),
    Input("input", "value")
)
def update_style(value):
    if int(value) < 0:
        color = "#E74C3C"
    elif int(value) == 0:
        color = "#F4D03F"
    elif int(value) > 0:
        color = "#2ECC71"
    return {
        "color": color
    }
```
